   import turtle

    def triangle(d):
        if d < 5:
            return    
        for _ in range(3):
            triangle(d/2)
            turtle.forward(d)
            
            turtle.left(120)
    
    turtle.tracer(0)
    turtle.speed(0)
    triangle(200)
    turtle.update()
    turtle.mainloop()
